```tsx
"use client";

import {
  Card,
  CardHeader,
  CardTitle,
  CardContent,
} from "@/components/ui/card";
import {
  MusicIcon,
  HandshakeIcon,
  SparklesIcon,
  FlameIcon,
  InfoIcon,
  TrendingUp,
  TrendingDown,
} from "lucide-react";
import { useQuery } from "@tanstack/react-query";
import { Tooltip, TooltipTrigger, TooltipContent, TooltipProvider } from "@/components/ui/tooltip";

/* ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ */
/* Generic Metric Card                           */
/* ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ */
interface MetricCardProps {
  title: string;
  value?: number | string;
  icon?: React.ReactNode;
  variant?: "default" | "success" | "warning";
  tooltip?: string;
  change?: number; // optional % change vs last week
}

export function MetricCard({
  title,
  value,
  icon,
  variant = "default",
  tooltip,
  change,
}: MetricCardProps) {
  const variantClasses = {
    default: "text-foreground",
    success: "text-green-400",
    warning: "text-yellow-400",
  }[variant];

  const trendColor =
    change && change !== 0
      ? change > 0
        ? "text-green-400"
        : "text-red-400"
      : "text-muted-foreground";

  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger asChild>
          <Card className="relative bg-muted/10 border border-muted/20 shadow-sm rounded-2xl hover:border-muted/40 transition-all cursor-default">
            <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
              <CardTitle className="text-sm font-medium text-muted-foreground flex items-center gap-1.5">
                {title}
                {tooltip && <InfoIcon className="h-3.5 w-3.5 opacity-60" />}
              </CardTitle>
              <span className={variantClasses}>{icon}</span>
            </CardHeader>

            <CardContent className="flex flex-col justify-between">
              <div className="text-3xl font-semibold tracking-tight">
                {value ?? "‚Äì"}
              </div>

              {change !== undefined && (
                <div
                  className={`flex items-center gap-1 mt-1 text-xs font-medium ${trendColor}`}
                >
                  {change > 0 ? (
                    <TrendingUp className="h-3 w-3" />
                  ) : change < 0 ? (
                    <TrendingDown className="h-3 w-3" />
                  ) : null}
                  {change === 0
                    ? "No change"
                    : `${change > 0 ? "+" : ""}${change.toFixed(1)}% vs last week`}
                </div>
              )}
            </CardContent>
          </Card>
        </TooltipTrigger>
        {tooltip && (
          <TooltipContent className="text-xs max-w-[200px] leading-snug">
            {tooltip}
          </TooltipContent>
        )}
      </Tooltip>
    </TooltipProvider>
  );
}

/* ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ */
/* Playlists Metrics                             */
/* ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ */
export function PlaylistMetrics() {
  const { data } = useQuery({
    queryKey: ["/api/metrics/playlists"],
    queryFn: () => fetch("/api/metrics/playlists").then((r) => r.json()),
  });

  return (
    <div className="grid gap-4 sm:grid-cols-3">
      <MetricCard
        title="Total Playlists"
        value={data?.totalPlaylists}
        icon={<MusicIcon size={18} />}
        tooltip="Total number of playlists currently tracked in your discovery network."
        change={data?.changePlaylists}
      />
      <MetricCard
        title="Avg Followers"
        value={data?.avgFollowers}
        icon={<SparklesIcon size={18} />}
        tooltip="Average follower count per playlist, showing reach of your discovery sources."
        change={data?.changeFollowers}
      />
      <MetricCard
        title="High-Impact Playlists"
        value={data?.highImpactPlaylists}
        icon={<FlameIcon size={18} />}
        variant="warning"
        tooltip="Playlists with an average track score ‚â• 7 ‚Äî hot zones for new signings."
        change={data?.changeHighImpact}
      />
    </div>
  );
}

/* ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ */
/* Tracks Metrics                                */
/* ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ */
export function TrackMetrics() {
  const { data } = useQuery({
    queryKey: ["/api/metrics/tracks"],
    queryFn: () => fetch("/api/metrics/tracks").then((r) => r.json()),
  });

  return (
    <div className="grid gap-4 sm:grid-cols-3">
      <MetricCard
        title="Total Enriched"
        value={data?.totalEnriched}
        icon={<MusicIcon size={18} />}
        tooltip="Number of tracks with full metadata enrichment (credits, socials, Chartmetric data)."
        change={data?.changeEnriched}
      />
      <MetricCard
        title="Deal-Ready Tracks"
        value={data?.dealReady}
        icon={<HandshakeIcon size={18} />}
        variant="success"
        tooltip="Tracks with high unsigned score (‚â•7) and valid contact ‚Äî ready for pub deal outreach."
        change={data?.changeDealReady}
      />
      <MetricCard
        title="Avg Unsigned Score"
        value={data?.avgScore}
        icon={<SparklesIcon size={18} />}
        variant="warning"
        tooltip="Average unsigned potential score across your active discovery pool."
        change={data?.changeAvgScore}
      />
    </div>
  );
}
```
