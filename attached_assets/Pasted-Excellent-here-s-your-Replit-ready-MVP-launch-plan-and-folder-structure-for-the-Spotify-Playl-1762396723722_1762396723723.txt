## ‚öôÔ∏è **Core Files and Roles**

### `/scripts/fetchPlaylists.ts`

Fetches tracks from your configured Spotify playlists.

```ts
import { spotify } from "@replit/spotify";
import fs from "fs";
import { DateTime } from "luxon";

import PLAYLISTS from "../config/playlists.json";

const week = DateTime.now().toISODate();

async function fetchPlaylists() {
  const data = [];

  for (const { id, name } of PLAYLISTS) {
    const playlist = await spotify.getPlaylist(id);
    for (const item of playlist.tracks.items) {
      const t = item.track;
      data.push({
        week,
        playlist: name,
        track_name: t.name,
        artist_name: t.artists.map(a => a.name).join(", "),
        spotify_url: t.external_urls.spotify,
        isrc: t.external_ids?.isrc || null,
        label: t.album.label || null,
        added_at: item.added_at,
      });
    }
  }

  fs.writeFileSync(
    `./data/snapshots/${week}.json`,
    JSON.stringify(data, null, 2)
  );

  console.log(`‚úÖ Saved ${data.length} tracks for ${week}`);
}

fetchPlaylists();
```

### `/scripts/scoreUnsigned.ts`

Implements the **Unsigned Likelihood Rubric**:

```ts
import fs from "fs";

const RUBRIC = {
  freshFinds: 3,
  drokit: 2,
  missingPublisher: 3,
  missingWriter: 2,
  partialPublisherShare: 1,
  majorLabel: -3,
};

function scoreTrack(t) {
  let score = 0;
  if (t.playlist.includes("Fresh Finds")) score += RUBRIC.freshFinds;
  if (t.label?.includes("DK")) score += RUBRIC.drokit;
  if (!t.publisher) score += RUBRIC.missingPublisher;
  if (!t.writer) score += RUBRIC.missingWriter;
  if (t.label && /(Sony|Warner|Universal)/i.test(t.label)) score += RUBRIC.majorLabel;
  return Math.max(0, Math.min(10, score));
}

function run() {
  const latest = fs
    .readdirSync("./data/snapshots")
    .sort()
    .reverse()[0];

  const data = JSON.parse(fs.readFileSync(`./data/snapshots/${latest}`));
  const scored = data.map(t => ({ ...t, unsigned_score: scoreTrack(t) }));
  fs.writeFileSync(`./data/snapshots/${latest.replace(".json", "_scored.json")}`,
    JSON.stringify(scored, null, 2));
  console.log(`üß† Scored ${scored.length} tracks`);
}

run();
```
