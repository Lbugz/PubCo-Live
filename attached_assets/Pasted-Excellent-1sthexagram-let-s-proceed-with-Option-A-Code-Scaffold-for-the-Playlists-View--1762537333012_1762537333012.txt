## üìÑ 2. `PlaylistsPage.tsx`

Main page route ‚Äî integrates filters, table, and drawer.

```tsx
import { useState } from "react";
import { PageHeader } from "@/components/ui/page-header";
import { StatsCards } from "@/components/ui/stats-cards";
import { PlaylistFilters } from "./PlaylistFilters";
import { PlaylistsTable } from "./PlaylistsTable";
import { PlaylistDrawer } from "./PlaylistDrawer";
import { usePlaylistsQuery } from "./usePlaylistsQuery";
import { Button } from "@/components/ui/button";
import { Plus } from "lucide-react";

export const PlaylistsPage = () => {
  const { data: playlists, isLoading } = usePlaylistsQuery();
  const [selectedPlaylist, setSelectedPlaylist] = useState(null);

  return (
    <div className="p-6 space-y-6">
      <PageHeader
        title="Playlists"
        actions={
          <Button variant="gradient" size="sm">
            <Plus className="mr-2 h-4 w-4" /> Add Playlist
          </Button>
        }
      />

      <StatsCards
        items={[
          { label: "Total", value: playlists?.length || 0, icon: "Library" },
          { label: "Active", value: playlists?.filter(p => p.status === "active").length || 0 },
          { label: "Paused", value: playlists?.filter(p => p.status === "paused").length || 0 },
          { label: "Error", value: playlists?.filter(p => p.status === "error").length || 0 },
        ]}
      />

      <PlaylistFilters />

      <PlaylistsTable
        playlists={playlists || []}
        isLoading={isLoading}
        onSelect={setSelectedPlaylist}
      />

      <PlaylistDrawer
        playlist={selectedPlaylist}
        onClose={() => setSelectedPlaylist(null)}
      />
    </div>
  );
};

export default PlaylistsPage;
```

## üìÑ 3. `PlaylistsTable.tsx`

```tsx
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Badge } from "@/components/ui/badge";
import { MoreHorizontal } from "lucide-react";

export const PlaylistsTable = ({ playlists, onSelect, isLoading }) => {
  if (isLoading) return <div className="text-gray-400">Loading playlists...</div>;

  return (
    <div className="glass-panel rounded-2xl overflow-hidden border border-white/10">
      <Table>
        <TableHeader>
          <TableRow>
            <TableHead>Playlist</TableHead>
            <TableHead>Source</TableHead>
            <TableHead>Tracks</TableHead>
            <TableHead>Curator</TableHead>
            <TableHead>Last Updated</TableHead>
            <TableHead>Status</TableHead>
            <TableHead></TableHead>
          </TableRow>
        </TableHeader>
        <TableBody>
          {playlists.map((p) => (
            <TableRow
              key={p.id}
              onClick={() => onSelect(p)}
              className="hover:bg-primary/5 cursor-pointer transition-colors"
            >
              <TableCell className="font-medium">{p.name}</TableCell>
              <TableCell>{p.source}</TableCell>
              <TableCell>{p.totalTracks}</TableCell>
              <TableCell>{p.curator}</TableCell>
              <TableCell>{p.lastSyncedAt || "‚Äî"}</TableCell>
              <TableCell>
                <Badge
                  variant={
                    p.status === "active"
                      ? "success"
                      : p.status === "paused"
                      ? "warning"
                      : "destructive"
                  }
                >
                  {p.status}
                </Badge>
              </TableCell>
              <TableCell>
                <MoreHorizontal className="h-4 w-4 text-gray-400" />
              </TableCell>
            </TableRow>
          ))}
        </TableBody>
      </Table>
    </div>
  );
};
```

## üìÑ 4. `PlaylistDrawer.tsx`

```tsx
import { Drawer, DrawerContent, DrawerHeader, DrawerTitle } from "@/components/ui/drawer";

  if (!playlist) return null;

  return (

        <DrawerHeader>
          <DrawerTitle className="flex items-center justify-between">
            {playlist.name}

        </DrawerHeader>

          <p><b>Curator:</b> {playlist.curator}</p>
          <p><b>Total Tracks:</b> {playlist.totalTracks}</p>
          <p><b>Last Synced:</b> {playlist.lastSyncedAt || "‚Äî"}</p>
```

## üìÑ 6. `usePlaylistsQuery.ts`

```ts
import { useQuery } from "@tanstack/react-query";

export const usePlaylistsQuery = () => {
  return useQuery({
    queryKey: ["playlists"],
    queryFn: async () => {
      const res = await fetch("/api/playlists");
      return res.json();
    },
  });
};
```
